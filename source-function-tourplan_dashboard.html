/**
 * TourPlan Dashboard Object class autoloader.
 * It locates and finds class via classes folder structure.
 *
 * @since 1.0.0
 *
 * @param string $class class name to be checked and autoloaded
 */
function tourplan_dashboard_autoload_class( $class ) {
    $parts = explode( '\\', $class );
    $name  = strtolower( str_replace( '_', '-', array_shift( $parts ) ) );
    if ( file_exists( TRPLN_PATH . 'classes/' . $name ) ) {
        if ( ! empty( $parts ) ) {
            $name .= '/' . implode( '/', $parts );
        }
        $class_file = TRPLN_PATH . 'classes/class-' . $name . '.php';
        if ( file_exists( $class_file ) ) {
            include_once $class_file;
        }
    } elseif ( empty( $parts ) && file_exists( TRPLN_PATH . 'classes/class-' . $name . '.php' ) ) {
        include_once TRPLN_PATH . 'classes/class-' . $name . '.php';
    }
}

/**
 * TourPlan Dashboard Helper to load and manipulate the overall instance.
 *
 * @since 1.0.0
 * @return  TourPlan_Dashboard  A single instance
 */
function tourplan_dashboard() {
    $request_data = array(
        'post'    => $_POST,
        'get'     => $_GET,
        'files'   => $_FILES,
        'request' => $_REQUEST,
        'server'  => $_SERVER,
    );

    // init Context.
    $instance = TourPlan_Dashboard::init();
    $instance->set_request_data( $request_data );
    return $instance;
}
